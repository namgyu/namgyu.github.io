import React from 'react';
import Stream from './Stream.react';
// import Collection from './Collection.react';
/**
 * @namespace
 */
class Application extends React.Component {
    /**
     * @description collectionTweets 초기화
     * @constructor
     * @method Application#constructor
     * @example
     * this.state = {
     *     collectionTweets: {}
     * };
     *
     */
    constructor(props) {
        super(props);
        this.state = {
            collectionTweets: {}
        };
    }

    /**
     * @description 트윗을 컬렉션에 추가
     * @method Application#addTweetToCollection
     * @param {Object} tweet 추가할 트윗 정보
     *
     */
    addTweetToCollection(tweet) {
        var collectionTweets = this.state.collectionTweets;
        collectionTweets[tweet.id] = tweet;
        this.setState({
            collectionTweets: collectionTweets
        });
    }
    
    /**
     * @description 전달받은 트윗을 id를 통해 컬렉션에서 삭제
     * @method Application#removeTweetFromCollection
     * @param {Object} tweet 삭제할 트윗 정보
     *
     */
    removeTweetFromCollection(tweet) {
        var collectionTweets = this.state.collectionTweets;
        delete collectionTweets[tweet.id];
        this.setState({
            collectionTweets: collectionTweets
        });
    }
    /**
     * @description 컬렉션에서 초기화
     * @method Application#removeAllTweetsFromCollection
     *
     */
    removeAllTweetsFromCollection() {
        this.setState({
            collectionTweets: {}
        });
    }

    render() {
        return(
            <div className="container-fulid">
                <div className="row">
                    <div className="col-md-4 text-center">
                        <Stream onAddTweetToCollection={this.addTweetToCollection} />
                    </div>
                    <div className="col-md-8">

                    </div>
                </div>
            </div>
        );
    }
}

export default Application;
